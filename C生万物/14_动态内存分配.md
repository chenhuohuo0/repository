##### 动态内存函数

###### malloc

```c
void* malloc(size_t size);

size - 需要的空间字节数<stdlib.h>
size=0场景c语言未定义，取决于编译器
开辟空间返回NULL
```

```c
int* p=(int*)malloc(10*sizeof(int));
if(p==NULL)
{
    priintf("%s\n",strerror(errno));//<string.h><errno.h>    
}
else
{
    //正常使用空间
    for(int i=0;i<10;i++)
    {
        *(p+i)=i;
    }
}
```



###### free

```c
当申请的空间不再使用时，把空间还给操作系统
void free(void* p);

free后p仍然指向原来的地址，空间释放后p应该置空
p=NULL;
如果p为NULL，free(p)不会做任何操作
```



###### calloc

```c
void* calloc(size_t num,size_t size);//<stdlib.h>and<malloc.h>
```

```c
在内存中为num个大小是size的元素开辟一块空间，把数据初始化为0，
开辟失败返回NULL
```

###### realloc

```c
void* realloc(void* memblock,size_t size);//<stdlib.h>and<malloc.h>
memblock指向需要调整的空间,size是重新开辟的空间字节数
```

```c
//调整动态开辟空间的大小
int* p=(int*)malloc(20);
if(p==NULL)
{
    printf("%s\n",strerror(errno));
}
else
{
    for(int i=0;i<5;i++)
    {
        *(p+i)=i;
    }
    //空间大小不对，用realloc调整
    int* p2=realloc(p,40);
    for(int i=5;i<10;i++)
    {
        *(p2+i)=i;
    }
    
    free(p2);//有借有还，再借不难
}

1.如果p后面有足够的空间，直接在后面追加20个字节，返回原来的p，
2.如果后面的空间不够，将寻找一块新地址开辟40个字节，将前20个字节(原数据)拷贝到新空间，再将原来的p释放，返回新地址p，此时p的值改变了
3.如果realloc开辟失败，返回NULL，这时直接赋值p2将会受到影响，应该用一个新指针判断一下
int *ptr=realloc(p,40);
if(ptr!=NULL)
{
    //...;
}
```

###### 常见的动态内存错误

```
1.每次调用动态内存函数进行判断是否为NULL，避免对空指针解引用操作
2.对动态开辟的空间越界访问，开辟的动态空间不够，越界访问时导致野指针问题
3.对非动态开辟的空间使用free
4.free只能释放起始位置的p，p变化后不能直接free
5.对同一块动态内存多次释放
6.动态开辟的空间忘记释放(内存泄漏)
```

###### 几个经典笔试题

```c
void getmemory(char* p)
{
    p=(char*)malloc(100);
}
void test(void)
{
    char* str=NULL;
    getmemory(str);
    strcpy(str,"hello world");
    printf(str);
}
int main()
{
    test();
}

//要用函数改变指针p的值，需要二级指针
//由于形参p值的丢失，malloc()无法free
```

```c
char* getmemory(void)
{
    char p[]="hello world";
    return p;
}
void test(void)
{
    char* str=NULL;
    str=getmemory();
    printf(str);
}
int main()
{
    test();
    return 0;
}

//p[]出范围就没了，导致非法访问返回的栈空间地址p，打印的值不可预知
```

```c
void getmemory(char** p,int num)
{
    *p=(char*)malloc(num);
}
void test(void)
{
    char* str=NULL;
    getmemory(&str,100);
    strcpy(str,"hello");
    printf(str);
}
int main()
{
    test();
    return 0;
}

//能够输出hello，但堆区仍然存在内存泄漏
```

```c
void test(void)
{
    char* str=(char*)malloc(100);
    strcpy(str,"hello");
    free(str)
    if(str!=NULL)
    {
        strcpy(str,"world");
        printf(str);
    }
}
int main()
{
    test();
    return 0;
}

//str被提前释放了，hello直接就没了，free释c放空间并不会把str置为null；仍然能通过str存储的地址拷贝world，但str就非法访问了
```

###### 柔性数组

```c
//结构体中最后一个成员允许是未知大小的数组，且大小是可调整的
struct S
{
    int n;//柔性数组成员前面至少有一个其他成员
    int arr[];//或者int arr[0]
}s;

printf("%d\n",sizeof(s));//4 - 计算大小时不包含柔性数组成员
struct S* ps=(struct S*)malloc(sizeof(struct S)+5*sizeof(int));
for(int i;i<5;i++)
{
    ps->arr[i]=i;//0 1 2 3 4
}
//调整大小
struct S* ptr=realloc(ps,44);
if(ptr!=NULL)
{
    for(int i=5;i<10;i++)
    {
        ptr->arr[i]=1;
    }
}
//释放
free(ptr);
```

```c
//法2实现柔性数组
struct S
{
    int n;
    int* arr;
};
struct S* ps=(struct S*)malloc(sizeof(struct S));
ps->arr=malloc(5*sizeof(int));
//调整大小
int *ptr=realloc(ps->arr,10*sizeof(int));
if(ptr!=NULL)
{
    //;
}
//释放
free(ps->arr);
free(ps);

//法2malloc两次，造成更多内存碎片，内存连续访问效率更高
//free两次且顺序不能乱，更容易出错
```
